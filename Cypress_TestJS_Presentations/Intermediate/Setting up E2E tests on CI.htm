CircleCI Cypress Orb

A much simpler CI configuration.

version: 2.1orbs:  # import Cypress orb by specifying an exact version x.y.z  # or the latest version 1.x.x using "@1" syntax  cypress: cypress-io/cypress@1workflows:  build:    jobs:      # "cypress" is the name of the imported orb      # "run" is the name of the job defined in Cypress orb      - cypress/run

version: 2.1orbs:  cypress: cypress-io/cypress@1workflows:  build:    jobs:      - cypress/install      - cypress/run:          requires:            - cypress/install          record: true # record results on Cypress Dashboard          parallel: true # split all specs across machines          parallelism: 4 # use 4 CircleCI machines
